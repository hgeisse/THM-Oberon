MODULE* KBDtest;

  IMPORT
    SYSTEM;

  CONST
    SP = 14;
    stackTop = 80000H;
    SB = 13;
    globVars = 1000H;
    serStat = -52;
    serData = -56;
    msData = -40;
    kbdData = -36;

  VAR
    x: INTEGER;

BEGIN
  SYSTEM.LDREG(SP, stackTop);
  SYSTEM.LDREG(SB, globVars);
  REPEAT
    REPEAT UNTIL SYSTEM.BIT(msData, 28);
    SYSTEM.GET(kbdData, x);
    REPEAT UNTIL SYSTEM.BIT(serStat, 1);
    SYSTEM.PUT(serData, x)
  UNTIL FALSE
END KBDtest.
